library(tidyverse)
library(themebg)

raw_train <- read_csv(
    "../data/raw/covid19-global-forecasting-week-1/train.csv",
    skip = 1,
    col_names = c(
        "id",
        "state",
        "country",
        "lat",
        "long",
        "date",
        "confirmed",
        "fatalities"
    )
)

raw_countryinfo <- read_csv("../data/raw/countryinfo/covid19countryinfo.csv") %>%
    mutate_at(
        c("quarantine", "schools", "restrictions"),
        parse_date,
        format = "%m/%d/%Y"
    )

df_info <- raw_countryinfo %>%
    filter(!is.na(pop))

df_country_totals <- raw_train %>%
    group_by(date, country) %>%
    summarize_at(c("confirmed", "fatalities"), sum, na.rm = TRUE)

df_day1 <- df_country_totals %>%
    filter(confirmed != 0) %>%
    group_by(country) %>%
    mutate(day = difftime(date, min(date), units = "days")) %>%
    mutate_at("day", as.numeric) %>%
    mutate_at("day", list(~if_else(country == "China", . + 22, . + 1)))

df_train <- df_country_totals %>%
    left_join(df_day1) %>%
    left_join(df_info, by = "country") %>%
    mutate(
        pop_10mil = pop / 10000000,
        confirm_pct = confirmed / pop,
        confirm_10mil = confirmed / pop_10mil,
        fatal_pct = fatalities / pop,
        fatal_10mil = fatalities / pop_10mil
    )

df_train_day <- df_train %>%
    filter(
        !is.na(day),
        pop >= 1000000
    )

df_curr <- df_train %>%
    group_by(country) %>%
    filter(
        date == max(date),
        pop >= 1000000
    ) %>%
    ungroup()

df_top25_confirm <- df_curr %>%
    top_n(25, confirmed)

df_top25_confirm_pct <- df_curr %>%
    top_n(25, confirm_pct)

df_top25_confirm_10mil <- df_curr %>%
    top_n(25, confirm_10mil)

df_top25_fatal <- df_curr %>%
    top_n(25, fatalities)

df_top25_fatal_pct <- df_curr %>%
    top_n(25, fatal_pct)

df_top25_fatal_10mil <- df_curr %>%
    top_n(25, fatal_10mil)

Exploratory Analysis

df_top25_confirm %>%
    arrange(desc(confirmed)) %>%
    mutate_at("country", as_factor) %>%
    mutate_at("country", fct_rev) %>%
    ggplot(aes(x = country, y = confirmed)) +
    geom_col() +
    coord_flip() +
    theme_bg()
